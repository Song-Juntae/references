    def 이상치처리(df, col, value):
        Q1 = df[col].quantile(0.25)
        Q3 = df[col].quantile(0.75)
        IQR = Q3 - Q1
        rev_range = 3
        if value < (Q1 - rev_range * IQR):
            value = Q1 - rev_range * IQR
        if value > (Q3 + rev_range * IQR):
            value = Q3 + rev_range * IQR
        return value
